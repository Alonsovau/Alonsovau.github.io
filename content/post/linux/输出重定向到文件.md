```bash

ls > log.log

ls >> log.log   --不覆盖log.log，追加

cat 1.txt 2> log.log   --错误输出到log.log

cat 1.txt > log.log 2>&1   --stderr也输出到stdout

stdin 0 <

stdout 1 1>

stderr 2 2>

```
